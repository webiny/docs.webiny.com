### `pagePathFromTitle`

**Constant** — imported from `webiny/admin/website-builder`

```typescript
import { pagePathFromTitle } from "webiny/admin/website-builder";
```

```typescript
export const pagePathFromTitle = (title: string = "") => {
  return slugify(title, {
    replacement: "-",
    lower: true,
    remove: /[*#\?<>_\{\}\[\]+~.()'"!:;@]/g,
    trim: false
  });
};
```
