Task Management Pattern:
```typescript
// Get single task
const result = await context.tasks.get(taskId);
if (result.isOk()) {
    const task = result.value;
    console.log(task.status, task.output);
}

// List tasks with filters
const listResult = await context.tasks.list({
    where: { name: "myTask", status: "pending" },
    limit: 10
});

// Update task
const updateResult = await context.tasks.update(taskId, {
    status: "aborted",
    error: { message: "Manual abort" }
});

// Delete task
const deleteResult = await context.tasks.delete(taskId);
```
